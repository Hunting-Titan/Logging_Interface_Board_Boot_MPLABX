73 void WL_SPS_V_INIT(void)
74
75 Summary:
76 This function initializes the WL_SPS_V and WL_SPS_V_SWITCH. It converts the the
77 * wl_sps_v from the 4 to 20mA output of the APM voltmeter to a voltage value.
78 ADC reading to mV.
79
80 Remarks:
81 Refer to the wl_sps_v.h interface header for function usage details.
82 */
83
84void WL_SPS_V_INIT(void)
85{
86 WL_SPS_V.new_voltage_flag = false; /* Initializes the variables.*/
87 WL_SPS_V.voltage = 0U;
88 WL_SPS_V.new_switch_value_flag = false;
89 WL_SPS_V.switch_voltage = 0U;
90 WL_SPS_V.switch_volt_constant = (float32_t)((3.3/4096.0)*1000); /* Constant to convert ADC counts into mVDC.*/
91 WL_SPS_V.current_constant = (float32_t)(((3.3/4096.0)/(24.9*6.0))*1000.0); /* Constant to ADC count into mA current. (24.9*6.0) accommodates sense resistor times gain.*/
92 WL_SPS_V.i_to_v_constant = (float32_t)(600/16); /* Constant to convert mA current to voltage.*/
93 WL_SPS_V.APM_voltage_flag = false;
94 WL_SPS_V.switch_voltage_min = 1000U;

98/*!*************************************************************************************
99 Function:
100 void Read_WL_SPS_V(void)
101
102 Summary:
103 This function reads the current and voltage A/D. It converts the the APM voltmeter
104 current to wl_sps voltage.
105
106 Remarks:
107 Refer to the wl_sps_v.h interface header for function usage details.
108 */
109
110void Read_WL_SPS_V(void)
111{
112 if(WL_SPS_V.new_voltage_flag == true) /* If new current value is available,*/
113 {
114 WL_SPS_V.new_voltage_flag = false; /* Reset the new value available flag.*/
115 /*LDRA_INSPECTED 96 S*/ /* Result is cast to an integer.*/
116 WL_SPS_V.WL_SPS_V_i = ((ADC.wl_sps_v * WL_SPS_V.current_constant)); /* Convert ADC counts to a uA current.*/
117 if(WL_SPS_V.WL_SPS_V_i >= 4.0F)
118 {
119 WL_SPS_V.voltage = rint((WL_SPS_V.WL_SPS_V_i - 4.0F)*
120 WL_SPS_V.i_to_v_constant); /* Convert the current reading to a voltage based on 0 to 600V scale.*/
121 }
122 else
123 {
124 WL_SPS_V.WL_SPS_V_i = 0.0F;
125 }
126 }
127 if(WL_SPS_V.new_switch_value_flag == true) /* If new voltage value is available,*/
128 {
129 /*LDRA_INSPECTED 96 S*/ /* Result is cast to an integer.*/
130 WL_SPS_V.switch_voltage =
131 (uint16_t)rint(ADC.wl_sps_switch * WL_SPS_V.switch_volt_constant); /* Convert ADC counts to a voltage.*/
132 WL_SPS_V.new_switch_value_flag = false; /* Reset the new value available flag.*/
133 if((WL_SPS_V.switch_voltage > WL_SPS_V.switch_voltage_min) &&
134 (WL_SPS_V.switch_voltage < WL_SPS_V.switch_voltage_max))
135 {
136 WL_SPS_V.APM_voltage_flag = true;
137 }
138 else
139 {
140 WL_SPS_V.APM_voltage_flag = false;
141 }
142 }
143}
