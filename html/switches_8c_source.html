817/*!*************************************************************************************
818 Function:
819 static void store_switches(void)
820
821 Summary:
822 This function stores the current switch positions to send back to the host when
823 requested.
824
825 Remarks:
826 Refer to the switches.h interface header for function usage details.
827 */
828static void store_switches(void)
829{
830 uint8_t cnt1,temp;
831
832 temp = 0U; /* Initialize the temporary switch position place holder.*/
833 for( cnt1 = 0U; cnt1 < 2U; cnt1++ ) /* The polarity switch position takes up the lower two bit positions*/
834 { /* of temp.*/
835 temp = (temp | SWITCH.pol_switch_S4[cnt1]) << 1;
836 }
837
838 for( cnt1 = 0U; cnt1 < 5U; cnt1++ ) /* The knob switch positions take up the upper five bit positions*/
839 { /* of temp. The MSB is not used.*/
840 temp = (temp | SWITCH.knob_switch_S3[cnt1]) << 1;
841 }
842 temp = temp >> 1; /* Shift back one to put bits in correct position.*/
843
844 SWITCH.store_buffer[ 0 ] = temp; /* Transfer the switch positions to store buffer [0].*/
845 temp = 0U; /* Reset the placeholder.*/
846 temp = (temp | SWITCH.ARM_CF_fire_switch_S8) << 1;
847 temp = (temp | SWITCH.DUMP_fire_switch_S7) << 1;
848 temp = (temp | SWITCH.ENABLE_switch_S1) << 1;
849 for( cnt1 = 0U; cnt1 < 3U; cnt1++ ) /* The key switch position takes up the next three bit positions*/
850 { /* of temp.*/
851 temp = (temp | SWITCH.key_switch_S6[cnt1]) << 1;
852 }
853 temp = temp | SWITCH.hvps_switch_S5 ;
854 SWITCH.store_buffer[ 1 ] = temp; /* Transfer the switch positions to store buffer [1]. Upper bit is not used.*/
855}
