392/*!*************************************************************************************
393 Function:
394 static void Decode_Manchester( void )
395
396 Summary:
397 Function to decode the Manchester response from a PerfSwitch.
398
399 Remarks:
400 Refer to the man.h interface header for function usage details.
401 */
402
403static void Decode_Manchester( void )
404{
405 uint8_t cnt;
406 uint8_t cnt2;
407 cnt = 0U; /* Counter for incoming buffer.*/
408 cnt2 = 0U; /* Counter for Manchester answer.*/
409
410 while( cnt2 < ( NoManBits + 2U ) ) /* While No of Manchester Bits + 2*/
411 { /* 2 extra transitions to accommodate the 1st (tickle) bit*/
412 /* of the pre-amble.*/
413 if( MAN.level[ cnt ] == 0x01U ) /* If the present level is a 1 then it must be a One Bit.*/
414 {
415 MAN.ans[ cnt2 ] = 0x01U; /* Set the answer to a one.*/
416 }
417 else /* Otherwise,*/
418 {
419 MAN.ans[ cnt2 ] = 0x00U; /* Set the answer to a zero.*/
420 }
421 if( MAN.time[ cnt + 1U ] == HalfBit ) /* If the time of next array element i.e.*/
422 { /* the other part of the waveform is a Half bit then*/
423 cnt++; /* increment the incoming buffer counter twice.*/
424 cnt++;
425 }
426 else /* Else increment only once.*/
427 {
428 cnt++;
429 }
430 cnt2++; /* Answer counter to be incremented by only one.*/
431 }
432}

434/*!*************************************************************************************
435 Function:
436 static void Package_Manchester( void )
437
438 Summary:
439 Manchester is decoded, the 1 X 32 answer array is packed into
440 a 4 X 8 bit message array.
441
442 Remarks:
443 Refer to the man.h interface header for function usage details.
444 */
445
446static void Package_Manchester( void )
447{
448 uint8_t cnt2; /* Counter for the Manchester answer array.*/
449
450 for( cnt2 = 0U; cnt2 < 8U; cnt2++ ) /* Package Manchester bits.*/
451 {
452 MAN.msg[ 0 ] = MAN.msg[ 0 ] | MAN.ans[ cnt2 + 1U ]; /* First byte of message array takes answer array elements 1 through 8.*/
453 MAN.msg[ 1 ] = MAN.msg[ 1 ] | MAN.ans[ cnt2 + 9U ]; /* Second byte of message array takes answer array elements 9 through 16.*/
454 MAN.msg[ 2 ] = MAN.msg[ 2 ] | MAN.ans[ cnt2 + 17U ]; /* Third byte of message array takes answer array elements 17 through 24.*/
455 MAN.msg[ 3 ] = MAN.msg[ 3 ] | MAN.ans[ cnt2 + 25U ]; /* Fourth byte of message array takes answer array elements 25 through 32.*/
456 if( cnt2 < 7U )
457 {
458 MAN.msg[ 0 ] = MAN.msg[ 0 ]<<1; /* Left-shift message elements to make room for next array answer element.*/
459 MAN.msg[ 1 ] = MAN.msg[ 1 ]<<1;
460 MAN.msg[ 2 ] = MAN.msg[ 2 ]<<1;
461 MAN.msg[ 3 ] = MAN.msg[ 3 ]<<1;
462 }
463 } /* Package Manchester bits Ends.*/
464}

590 }

592

596 }

598 }
