454 Function receives a command for validation and returns a true if it is
455 a valid command.
456
457 Remarks:

460
461bool Valid_Command( uchar8_t msg )
462{
463 bool ans; /* Returned value from check.*/
464
465 const uchar8_t CommandBitMap[NoOfCommands] ={0x65U,0x66U,0x67U,0x68U,0x69U,0x6AU,
466 0x6BU,0x6CU,0x6DU,0x6EU,0x6FU,0x70U};
467
468 ans = false; /* Set response to false.*/
469 if ((strchr( ( char * ) CommandBitMap, msg ) != NULL ) && ( msg != 0U )) /* Perform a compare to look for the command in the list.*/
470 {
471 ans = true; /* If we found a match, set to true.*/
472 }
473 else /* Otherwise,*/
474 {
475 ans = false; /* set to false.*/
476 }
477 return ans;
478}
