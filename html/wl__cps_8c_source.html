225/*!*************************************************************************************
226 Function:
227 static Read_WL_CPS_V_I(void)
228
229 Summary:
230 This task reads the current and voltage A/D. It stores the max current value.
231
232 Remarks:
233 Refer to the wl_cps.h interface header for function usage details.
234 */
235
236static void Read_WL_CPS_V_I(void)
237{
238 uint8_t count; /* counter to create average current.*/
239
240 if(WL_CPS.new_voltage_flag == true) /* If new voltage value is available,*/
241 {
242 /*LDRA_INSPECTED 96 S*/ /* Result is cast to an integer.*/
243 WL_CPS.voltage = (uint16_t)rint(ADC.wl_cps_v * WL_CPS.volt_constant); /* Convert ADC counts to a voltage.*/
244 WL_CPS.new_voltage_flag = false; /* Reset the new values available flag.*/
245 }
246 if(WL_CPS.new_current_flag == true) /* If new current value is available,*/
247 {
248 WL_CPS.new_current_flag = false; /* Reset the new values available flag.*/
249 /*LDRA_INSPECTED 96 S*/ /* Result is cast to an integer.*/
250 WL_CPS.current = (uint16_t)rint((ADC.wl_cps_i * WL_CPS.current_constant)); /* Convert ADC counts to a mA current.*/
251 /*LDRA_INSPECTED 130 D*/ /* WL_CPS.current does not need to be checked.*/
252 WL_CPS.current_array[WL_CPS.avg_count] = WL_CPS.current; /* Place new current reading into array.*/
253 WL_CPS.avg_current = 0U; /* Reset the average current result.*/
254
255 for(count = 0U; count < WL_CPS.avg_count_max; count++) /* Create the sum of the current readings.*/
256 {
257 WL_CPS.avg_current = WL_CPS.avg_current + WL_CPS.current_array[count];
258 }
259 WL_CPS.avg_current = WL_CPS.avg_current / WL_CPS.avg_count_max; /* Calculate the average of the readings.*/
260 WL_CPS.avg_count++; /* Increment the array counter.*/
261
262 if(WL_CPS.avg_count == WL_CPS.avg_count_max) /* If the counter has reached the end of the array,*/
263 {
264 WL_CPS.avg_count = 0U; /* reset the counter.*/
265 }
266 }
267
268 if(WL_CPS.avg_current > WL_CPS.max_current) /* Check for a maximum current reading.*/
269 {
270 WL_CPS.max_current = WL_CPS.avg_current;
271 }
272}

277
278 Summary:

282 Remarks:

285
286static void Check_WL_CPS_Over_Current(void)
287{
288 if((WL_CPS.current >= WL_CPS.current_limit)) /* WL_CPS Current has exceeded current limit,*/
289 {
290 WL_CPS.over_current_count ++; /* Increment occurrence count.*/
291 if(WL_CPS.over_current_count > 5U) /* If it exceeds limit 5 times in a row,*/
292 {
293// WL_CPS_SWOff(); /* turn the WL_CPS supply off.*/
294 Set_Status(14); /* Set the status bit to indicate an over current.*/
295 WL_CPS.max_current = WL_CPS.current;
296 }
297 }
298 else
299 {
300 WL_CPS.over_current_count = 0U; /* Reset Count.*/
301 }
302}
