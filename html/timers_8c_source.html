94{
95 switch ( TIMER.state ) /* Check the TIMERS current state.*/
96 {
97 case TIMER_STATE_INIT: /* TIMERS initial state.*/
98 {
99 TIMER.tmrDrvHandle = DRV_HANDLE_INVALID; /* Don't use until TIMER instance 1 is opened.*/
100 TIMER.state = TIMER_STATE_TIMER_START; /* Move on to start the timer and initialize counts.*/
101 }
102 break;
103
104 case TIMER_STATE_TIMER_START:
105 {
106 if(TIMER.tmrDrvHandle == DRV_HANDLE_INVALID) /* If Timer instance 1 has not been opened,*/
107 {
108 TIMER.tmrDrvHandle = DRV_TMR_Open ( DRV_TMR_INDEX_1,
109 DRV_IO_INTENT_EXCLUSIVE ); /* Open Timer instance 1. It is using Timer 7. Set for 100uS.*/
110 }
111
112 if ( TIMER.tmrDrvHandle != DRV_HANDLE_INVALID ) /* If Timer instance 1 is open,*/
113 {
114 (void)DRV_TMR_Start(TIMER.tmrDrvHandle); /* Start the timer.*/
115
116 TIMER.timer1_100uS_tick = false; /* Initialize timer1 100uS Timer Tick to false.*/
117 TIMER.timer1_1mS_tick = false; /* Initialize timer1 1mS Timer Tick to false.*/
118 TIMER.timer1_10mS_tick = false; /* Initialize timer1 10mS Timer Tick to false.*/
119 TIMER.timer1_100mS_tick = false; /* Initialize timer1 100mS Timer Tick to false.*/

122 TIMER.timer1_100uS_count = 0U; /* Initialize timer1 100uS count to 0.*/

126 TIMER.timer1_1000mS_count = 0U; /* Initialize timer11 1000mS count to 0.*/
127
128 TIMER.state = TIMER_STATE_SERVICE_TASKS; /* Move on to perform the Timer tasks.*/
129 }
130 }
131 break;
132
133 case TIMER_STATE_SERVICE_TASKS:
134 {
135 if (TIMER.timer1_100uS_tick == true) /* Check if interrupt changed Timer1_100uS_Tick flag.*/

138// HV_PS.tick = true; /* Flag to allow the HV_PS control voltage to be adjusted.*/

140// FSK.tick = true;
141 GPT.tick = true;

145 TIMER.timer1_100uS_count = 0U; /* reset the counter and*/
146 TIMER.timer1_1mS_tick = true; /* set the 1mS tick to true to indicate 1mS has elapsed.*/
147 }
148 }
149

155// GAM.tick = true;
156 TIMER.timer1_1mS_count++; /* Increment the 1mS counter.*/

158 {
159 TIMER.timer1_1mS_count = 0U; /* reset the counter and*/
160 TIMER.timer1_10mS_tick = true; /* set the 10mS tick to true to indicate 10mS has elapsed.*/

162 }
163

167// MAN.tick = true; /* Set to true so the Manchester timer can be incremented.*/

169// appData.hidDataReceived = true; /* Set this to true to initiate a data transfer.*/

171// appData.state = HID_STATE_TRANSMIT;
172 HIGH_SIDE.tick = true;
173 appData.timer_tick = true;

176 REPORT.data[5] = ((TIMER.count & 0x00FF0000U)>> 16);
177 REPORT.data[6] = ((TIMER.count & 0x0000FF00U)>> 8);

220 TIMER.timer1_1000mS_count = 0U; /* reset the counter.*/
