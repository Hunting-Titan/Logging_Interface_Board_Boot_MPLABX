81void CF_Tasks ( void )
82{
83 switch(CF.state)
84 {
85 case CF_STATE_INIT:
86 {
87 CF.upper_current_limit = 2000; /* Set to 2000mA.*/
88 CF.current_limit = 2000; /* Set the initial current limit to 2000mA.*/
89 CF.tick = false;
90 CF.overcurrent_flag = false; /* Initialize the overcurrent flag to false.*/
91 CF.new_current_values_flag = false; /* Initialize the new values available flag as false.*/
92 CF.sensor_constant = /* ADC reading to mA current. Convert to integer causes @ 1% error.(16).*/
93 (uint16_t)(rint((3.3F/4096.0F)*(1.0F/(0.066F*1.3243F))*1000.0F)); /* Convert ADC reading to mA current. Convert to integer causes @ 2% error.(9).*/
94 /* 1.3243 is circuit gain. 0.066 is hall effect IC sensitivity.*/
95 CF.state = CF_SENSOR_OFFSET; /* Move on to record the zero current sensor offset.*/
96 CF.sensor_offset = 0U;
97 CF.sensor_offset_tick = false;
98 CF.avg_count = 0U; /* Initialize the current array counter.*/
99 CF.avg_count_max = 5; /* Set the number of current readings for array.*/
100 (void) memset(CF.current_array, 0, (CF.avg_count_max*2U)); /* Initialize the current array values to zero.*/
101 CF.avg_current = 0U; /* Initialize the average current result.*/
102 }
103 break;
104
105 case CF_SENSOR_OFFSET:
106 {
107 if(((SW_COM.S3_state == RLY_ARM_CF) || (SW_COM.S3_state == RLY_ARM_CF_AUTO))/* If the knob switch is in the ARM_CF or the AUTO_ARM_CF position and the sensor*/
108 && (CF.sensor_offset_tick == true)) /* offset one second timer has completed,*/
109 { /* The one second timer allows the readings to settle during initialization.*/
110 if((CF.new_current_values_flag == true) ) /* and a new value is available,*/
111 { /* The one second timer allows the readings to settle at power up.*/
112 CF.sensor_offset = ADC.wl_sps_i_cf; /* First reading is the zero current sensor offset.*/
113 CF.state = CF_STATE_READ_I; /* Move on to read the CF current.*/
114 CF.new_current_values_flag = false; /* Reset the new current value flag.*/
115 }
116 else
117 {
118 /*Comment.*/
119 }
120 }
121 CF.sensor_offset_tick = false; /* Reset the offset_timer tick.*/
122 }
123 break;
124
125 case CF_STATE_READ_I:
126 {
127 uint8_t count; /* counter to create average current.*/
128
129 if(CF.new_current_values_flag == true) /* If a new value is available, check*/
130 {
131 if(ADC.wl_sps_i_cf >= CF.sensor_offset) /* if the ADC reading is greater than the zero current offset reading,*/
132 {
133 CF.current = ADC.wl_sps_i_cf - CF.sensor_offset; /* Determine the counts for a reading greater than the offset.*/
134 }
135 else /* Otherwise,*/
136 {
137 CF.current = CF.sensor_offset - ADC.wl_sps_i_cf; /* determine the counts for a reading less than the offset.*/
138 }
139 /*LDRA_INSPECTED 130 D*/ /* CF.sensor_constant returns as integer. No check needed.*/
140 CF.current = CF.current * CF.sensor_constant; /* Convert ADC counts to mA. The current is an absolute value. No polarity indication.*/
141
142 CF.current_array[CF.avg_count] = CF.current; /* Place new current reading into array.*/
143 CF.avg_current = 0U; /* Reset the average current result.*/
144
145 for(count = 0U; count < CF.avg_count_max; count++) /* Create the sum of the current readings.*/
146 {
147 CF.avg_current = CF.avg_current + CF.current_array[count];
148 }
149 CF.avg_current = CF.avg_current / CF.avg_count_max; /* Calculate the average of the readings.*/
150 CF.avg_count++; /* Increment the array counter.*/
151
152 if(CF.avg_count == CF.avg_count_max) /* If the counter has reached the end of the array,*/
153 {
154 CF.avg_count = 0U; /* reset the counter.*/
155 }
156
157 if(CF.current > CF.max_current) /* Check for a maximum current reading.*/
158 {
159 CF.max_current = CF.current;
160 }
161 CF.new_current_values_flag = false; /* Reset the new current value flag.*/
162 CF.state = CF_STATE_CHECK_OVERCURRENT; /* Move on to the overcurrent check.*/
163 }
164 }
165 break;
166
167 case CF_STATE_CHECK_OVERCURRENT:
168 {
169 if((CF.current > CF.current_limit)) /* CF Current has exceeded current limit,*/
170 {
171 CF.over_current_count ++; /* Increment occurrence count.*/
172 if(CF.over_current_count > 5U) /* If it exceeds limit 5 times in a row,*/
173 {
174 if(AUTO_CF.ignore_shot_flag == false) /* If we are not ignoring shot shot detection,*/
175 {
176 CF.overcurrent_flag = true; /* Set flag to true. Used in HVPS.c*/
177 Set_Status(7); /* Set status bit to indicate WL_SPS overcurrent.*/
178 }
179 }
180 }
181 else
182 {
183 CF.over_current_count = 0U; /* Reset Count.*/
184 }
185 CF.state = CF_STATE_READ_I; /* move on to read another current.*/
186 }
187 break;
188
189 default:
190 {
191 /*Comment.*/
192 }
193 break;
194 }
195}
