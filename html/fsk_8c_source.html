101 Function:
102 void FSK_Tasks( ( void )
103
104 Summary:
105 This routine is the FSK tasks function. It defines the (FSK) state machine and
106 core logic.
107
108 Remarks:
109 Refer to the fsk.h interface header for function usage details.
110 */
111
112void FSK_Tasks( void )
113{
114 switch( FSK.state)
115 {
116 case FSK_STATE_INIT:
117 {
118 Init_FSK( ); /* Initialize the FSK parameters and start Timer 2.*/
119 FSK.state = FSK_STATE_CHECK_MESSAGE_READY; /* Move on to see if a message to a PerfSwitch is ready to be sent.*/
120 PRE_MSG.state = PREAMBLE_STATE_IDLE; /* Put the FSK preamble state machine in idle.*/
121 MSG.state = MSG_STATE_IDLE; /* Put the Send_Message_task state machine in idle.*/
122 }
123 break;
124
125 case FSK_STATE_CHECK_MESSAGE_READY:
126 {
127 if( DWNMSG.xmit_ready_flag != 0x00U ) /* If we are ready to transmit a message to a PerfSwitch.*/
128 { /* This flag is set in the Protocol B commands.*/
129 FSK.state = FSK_STATE_PREP_MESSAGE; /* Move on to prep the message.*/
130 }
131 }
132 break;
133
134 case FSK_STATE_PREP_MESSAGE:
135 {
136 switch( DWNMSG.msg_length ) /* Determine the message length.*/
137 {
138 case 0x04U:
139 {
140 MSG.command = DWNMSG.identifier; /* Configure the message command.*/
141 MSG.length = DWNMSG.msg_length - 1U; /* Configure the message length.*/
142 FSK.state = FSK_STATE_SEND_MESSAGE; /* Move on to send the message.*/
143 MSG.state = MSG_STATE_CAL_MAN_BIAS_1; /* Initialize the first task in the SEND_MESSAGE_Tasks machine.*/
144 }
145 break;
146
147 case 0x14U:
148 {
149 Reset_CRC_Value( ); /* Reset the CRC value.*/
150 Calc_CRC( 3, DWNMSG.identifier ); /* CRC is not calculated for the Long/Short Bit */
151 Calc_CRC( 8, DWNMSG.command ); /* Update the CRC result with the command byte.*/
152 FSK.crc = Get_CRC_Value( );
153
154 MESSAGE.byte[ 2 ]= DWNMSG.identifier; /* Prepare the message.*/
155 MESSAGE.byte[ 1 ]= DWNMSG.command;
156 MESSAGE.byte[ 0 ]= FSK.crc;
157
158 MSG.command = MESSAGE.dword; /* Configure the message command.*/
159 MSG.length = DWNMSG.msg_length - 1U; /* Configure the message length.*/
160 FSK.state = FSK_STATE_SEND_MESSAGE; /* Move on to send the message.*/
161 MSG.state = MSG_STATE_CAL_MAN_BIAS_1; /* Initialize the first task in the SEND_MESSAGE_Tasks machine.*/
162
163 }
164 break;
165
166 default:
167 {
168 /*Comment.*/
169 }
170 break;
171 }
172 }
173 break;
174
175 case FSK_STATE_SEND_MESSAGE:
176 {
177 Send_Message_Tasks(); /* Send the FSK message to the PerfSwitch.*/
178 if(FSK.send_message_complete_flag == true) /* If we have completed sending the message,*/
179 {
180 FSK.state = FSK_STATE_WAIT; /* Move on to wait for the Manchester response.*/
181 MAN.state = MAN_STATE_RECEIVE_START; /* Initiate the Manchester Receive tasks.*/
182 }
183 }
184 break;
185
186 case FSK_STATE_WAIT: /* Wait here for the Manchester response.*/
187 {
188
189 }
190 break;
191
192 case FSK_STATE_COMPLETE_MESSAGE: /* Machines moves here after Manchester response is received.*/
193 {
194 if(MAN.trynumber <= 0x02U) /* If we do not have to retry to send an FSK message,*/
195 {
196 DWNMSG.xmit_ready_flag = 0x00U; /* Reset the transmit ready flag.*/
197 MESSAGE.dword = 0U; /* Reset the message.*/
198 FSK.state = FSK_STATE_CHECK_MESSAGE_READY; /* Move on to see if another message needs to be transmitted.*/
199 FSK.send_message_complete_flag = false;
200 }
201 else /* Otherwise,*/
202 {
203 FSK.state = FSK_STATE_CHECK_MESSAGE_READY; /* Move on to retransmit the message.*/
204 FSK.send_message_complete_flag = false;
205 }
206 }
207 break;
208
209 default:
210 {
211 /*Comment.*/
212 }
213 break;
214 }

362static void Send_Message_Tasks(void )
363{
364 switch(MSG.state)
365 {
366 case MSG_STATE_IDLE: /* The initial state to wait for a FSK message to send.*/
367 {
368 /* No action occurs here.*/
369 }
370 break;
371
372 case MSG_STATE_CAL_MAN_BIAS_1: /* Perform the first Manchester bias adjustment.*/
373 { /* Accessed from the FSK state machine.*/
374 if(MAN.manual_bias_flag == false) /* If we have not set the MAN bias manually,*/
375 {
376// if((MAN.spi_sent_flag == false) && /* If we have not sent the command and the MAN_BIAS SPI port is open,*/
377// (MAN_BIAS.spi_bias_fsk != DRV_HANDLE_INVALID))
378 {
379 Calc_Auto_Bias(); /* Calculate the correct bias and send data to the MAN_BIAS SPI port.*/
380 MAN.spi_sent_flag = true; /* Set this flag to show we sent data to the MAN_BIAS SPI port.*/
381 }
382
383 if(MAN.spi_write_complete_flag == true) /* If the SPI port is through sending data,*/
384 {
385 MAN.spi_write_complete_flag = false; /* reset the flag.*/
386 MSG.state = MSG_STATE_SEND_PREAMBLE; /* Move on send the message preamble.*/
387 PRE_MSG.state = PREAMBLE_STATE_SEND_TICKLE; /* Set the PRE_MSG state machine to its initial state.*/
388 MAN.spi_sent_flag = false; /* Reset the flag so we can send data.*/
389 }
390 }
391 }
392 break;
393
394 case MSG_STATE_SEND_PREAMBLE: /* This state contains the PRE_MSG state machine.*/
395 {
396 switch(PRE_MSG.state) /* This machine controls the sending of the preamble bits.*/
397 {
398 case PREAMBLE_STATE_IDLE: /* Wait here until the next message needs to be sent.*/
399 {
400 /* No action occurs here.*/
401 }
402 break;
403
404 case PREAMBLE_STATE_SEND_TICKLE: /* Send the first MARK 'tickle' bit.*/
405 {
406 Send_Mark();
407 PRE_MSG.state = PREAMBLE_STATE_NULL; /* Move on to wait for the null period.*/
408 }
409 break;
410
411 case PREAMBLE_STATE_NULL: /* Null time between tickle bit and remaining preamble.*/
412 {
413 if(FSK.tick == true) /* If 100uS FSK tick is true,*/
414 {
415 FSK.tick = false; /* reset the tick.*/
416 if(FSK.null_count == 25U) /* If 2.5mS has passed,*/
417 {
418 FSK.null_count = 0U; /* reset the count.*/
419 PRE_MSG.state = PREAMBLE_STATE_SEND_MARK_1; /* Move on to send the first MARK bit.*/
420 }
421 else /* Otherwise,*/
422 {
423 FSK.null_count = FSK.null_count + 1U; /* increment the null timer count.*/
424 }
425 }
426 }
427 break;
428
429 case PREAMBLE_STATE_SEND_MARK_1: /* Send the first MARK bit of the preamble.*/
430 {
431 Send_Mark();
432 PRE_MSG.state = PREAMBLE_STATE_SEND_MARK_2; /* Move on to send the second MARK bit.*/
433 }
434 break;
435
436 case PREAMBLE_STATE_SEND_MARK_2: /* Send the second MARK bit of the preamble.*/
437 {
438 Send_Mark();
439 PRE_MSG.state = PREAMBLE_STATE_SEND_SPACE; /* Move on to send the first SPACE bit.*/
440 }
441 break;
442
443 case PREAMBLE_STATE_SEND_SPACE: /* Send the SPACE bit of the preamble.*/
444 {
445 Send_Space();
446 PRE_MSG.state = PREAMBLE_STATE_IDLE;
447 MSG.state = MSG_STATE_SEND_MESSAGE;
448 }
449 break;

454 }
455 break;
456 }
457 }
458 break;

460 case MSG_STATE_SEND_MESSAGE:
461 {
462 int8_t bit_position;
463
464 for(bit_position = (int8_t)MSG.length; bit_position >= 0;bit_position--) /* For loop required to maintain proper FSK timing to PerfSwitch.*/
465 { /* This cannot be set up like the preamble. This is sent MSB first.*/
466 /*LDRA_INSPECTED 403 S*/ /* bit_position will not be a negative number.*/
467 if( testbit( MSG.command, bit_position ) == 0x01U ) /* If the bit in the message is a one,*/
468 {
469 Send_Mark( ); /* send an FSK one.*/
470 }
471 else /* Otherwise,*/
472 {
473 Send_Space( ); /* send an FSK zero.*/
474 }
475 }
476
477 Send_Mark( ); /* Send an extra Mark (Stop bit)or else PerfSwitch won't respond.*/
478 MSG.state = MSG_STATE_NULL; /* Move on to wait for a null time.*/
479 }
480 break;
481

483 {

485 {
486 FSK.tick = false; /* reset the tick.*/
487 if(FSK.null_count == 25U) /* If 2.5mS has passed,*/

491 FSK.send_message_complete_flag = true; /* Flag to indicate that we have completed sending the FSK message.*/
492 }
493 else /* Otherwise,*/
494 {
495 FSK.null_count = FSK.null_count + 1U; /* increment the null timer count.*/
496 }
497 }
498 }
499 break;

502 {
503 /*Comment.*/
504 }
505 break;
506 }
