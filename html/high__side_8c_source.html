90void HIGH_SIDE_Tasks ( void )
91{
92 switch(HIGH_SIDE.state)
93 {
94 case HIGH_SIDE_STATE_INIT:
95 {
96 if(SWITCH.command_setting == 0U) /* If we received a command to read the High Side data,*/
97 {
98 HIGH_SIDE.write_complete = false; /* Set this flag to show we have not completed a write to the host.*/
99 HIGH_SIDE.usage = false; /* Flag to show we are not communicating through the UART.*/
100 HIGH_SIDE.handleUSART1 = DRV_HANDLE_INVALID; /* Set the UART handle to invalid.*/
101 HIGH_SIDE.state = HIGH_SIDE_STATE_OPEN; /* Move on to open the UART port.*/
102 HIGH_SIDE.bytes_received = 0U;
103 HIGH_SIDE.checksum = 0U;
104 HIGH_SIDE.byte_count = 0U;
105 HIGH_SIDE.timeout_count = 0U;

109 Clear_Status( 6 ); /* Clear the status bit to show we received good data.*/

111 }
112 break;

114 case HIGH_SIDE_STATE_OPEN:
115 {
116 if (HIGH_SIDE.handleUSART1 == DRV_HANDLE_INVALID) /* Open the UART. This port is used for debugging only.*/
117 {
118 /*LDRA_INSPECTED 123 S*/
119 /*LDRA_INSPECTED 120 S*/ /* MPLAB Harmony created function.*/
120 HIGH_SIDE.handleUSART1 = DRV_USART_Open(1,
121 DRV_IO_INTENT_READWRITE|DRV_IO_INTENT_NONBLOCKING);
122 }
123 if(HIGH_SIDE.handleUSART1 != DRV_HANDLE_INVALID) /* Once the UART port is open,*/

138 Nop();
139 DRV_USART_Close(HIGH_SIDE.handleUSART1);
140 HIGH_SIDE.state = HIGH_SIDE_STATE_CHECKSUM;
141 HIGH_SIDE.timeout_count = 0U;
142 }
143 else if ((HIGH_SIDE.bytes_received == HIGH_SIDE.data[2U]) && /* Otherwise if the bytes received are 'Tool Data',*/
144 (HIGH_SIDE.data[1U] == 3U))
145 {
146 Nop();
147 DRV_USART_Close(HIGH_SIDE.handleUSART1);
148 HIGH_SIDE.state = HIGH_SIDE_STATE_CHECKSUM;

150 }
151 else
152 {
153 /**/
154 }
155 if((HIGH_SIDE.bytes_received == 0U) && (HIGH_SIDE.tick == true)) /* If data is not being received,*/
156 {
157 HIGH_SIDE.tick = false; /* Reset the timer tick.*/
158 HIGH_SIDE.timeout_count = HIGH_SIDE.timeout_count + 1U; /* Increment the timer count.*/
159 if(HIGH_SIDE.timeout_count == 21U) /* If 210mS have gone by,*/
160 {
161 Set_Status( 6 ); /* Set the status bit to show the data is bad.*/
162 HIGH_SIDE.timeout_count = 0U; /* Reset the time out count.*/

179 { /* State will move to WRITE from commmodule.c.*/
180

185 {
186 for(HIGH_SIDE.byte_count=0; /* Calculate the checksum as a modulo 256.*/
187 HIGH_SIDE.byte_count < HIGH_SIDE.bytes_received - 1U;
188 HIGH_SIDE.byte_count++)
189 { /* Add all of the data values except the checksum*/
190 HIGH_SIDE.checksum = HIGH_SIDE.checksum + /* and use the LSB for comparison.*/
191 HIGH_SIDE.data[HIGH_SIDE.byte_count];
192 }
193 if(HIGH_SIDE.checksum == HIGH_SIDE.data[HIGH_SIDE.bytes_received - 1U]) /* If the checksum is valid,*/
194 {
195 Clear_Status( 6 ); /* Clear the status bit to show we received good data.*/
196 for(HIGH_SIDE.byte_count = 0U;

206 }

208 {

210 Set_Status( 6 ); /* Set the status bit to show we did not receive good data.*/

214 HIGH_SIDE.byte_count ++)

218 HIGH_SIDE.bytes_received = 0U;

221 }

224 case HIGH_SIDE_STATE_WRITE: /* Send the command response to the host.*/
225 {
226 WriteHSUART(); /* Send the command response.*/
227 if(HIGH_SIDE.write_complete == true) /* If we have completed sending the response,*/
228 {
229 HIGH_SIDE.write_complete = false; /* reset the flag.*/
230 HIGH_SIDE.state = HIGH_SIDE_STATE_READ; /* Move to look for the next data from the host.*/
231 HIGH_SIDE.usage = false; /* Flag to show we are not communicating through the UART.*/
232 }
233 }
234 break;

236 default:
237 {
238 /*Comment.*/
239 }

261static void ReadHSUART(void)
262{
263 static i = 0U;
264 static uint8_t uart_bytes=0U;
265
266 /*LDRA_INSPECTED 433 S*/ /* MPLAB Harmony function returns a boolean result.*/
267 if(DRV_USART_ReceiverBufferIsEmpty(HIGH_SIDE.handleUSART1) == false) /* If the UART receive buffer has data in it,*/
268 {
269 /*LDRA_INSPECTED 434 S*/ /*MPLAB Harmony function.*/
270
271 uart_bytes = DRV_USART_ReadByte(HIGH_SIDE.handleUSART1); /* Read a received byte and */
272 if((HIGH_SIDE.bytes_received == 0U) && (uart_bytes == 0xAC)) /* If the first byte received is the sync byte,*/
273 {
274 HIGH_SIDE.data[HIGH_SIDE.bytes_received] = uart_bytes; /* Put the byte into the data array.*/
275 HIGH_SIDE.usage = true; /* Shows that we received data via the UART and not the USB.*/
276 HIGH_SIDE.bytes_received = HIGH_SIDE.bytes_received + 1U; /* Increment counter for number of bytes received via the UART.*/
277 }
278 else if((HIGH_SIDE.bytes_received > 0U)) /* If we have already received the sync byte,*/
279 {
280 HIGH_SIDE.data[HIGH_SIDE.bytes_received] = uart_bytes; /* Put the byte into the data array.*/
281 HIGH_SIDE.usage = true; /* Shows that we received data via the UART and not the USB.*/

291 }
