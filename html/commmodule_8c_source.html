92void COMM_Tasks(void)

94 switch ( COMM.state ) /* Check the machines current state.*/
95 {
96 case COMM_STATE_INIT:

100 COMM.rx_count = 0U; /* Set the receive counter to 0.*/

102 COMM.state = COMM_STATE_IDLE; /* Move on to wait for a host communication.*/
103 }
104 break;
105
106 case COMM_STATE_VALIDATE: /* Comes here when we receive a byte from the UART.*/
107 {
108 if( COMM.processing == false ) /* If no command is currently being processed, */
109 {
110 ValidateComm(); /* see if we received a valid command.*/
111 }
112 if( COMM.processing == true ) /* If we received a valid command,*/
113 {
114 COMM.state = COMM_STATE_WAIT; /* Move on to wait for command to be executed.*/
115 CMD.process_complete_flag = false; /* Flag to show a command needs to be executed.*/

171
172static void ValidateComm( void )
173{
174 uchar8_t temp;
175
176 temp = Fifo_Get( &RX_FIFO ); /* Get the first byte and check to see if it is a valid data word.*/
177
178 if( Valid_Command( temp ) == false ) /* If the command is not valid, set the status byte and clear the RxFifo.*/
179 {
180 Set_Status( 15 ); /* Set the Status byte to show an invalid command was received.*/
181 Prepare_Return_A( 0x5C, 0, 0 ); /* Prepare the return with a Status query 0x5C message.*/
182 while(COMM.rx_count < CMD.cmd_size) /* While the the receive byte counter is less than the command size,*/
183 {
184 ( void ) Fifo_Get( &RX_FIFO ); /* Clear the FIFO Out & start over.*/
185 COMM.rx_count = COMM.rx_count + 1U; /* Increment the counter.*/
186 }
187 }
188 else
189 {
190 Clear_Status( 15 ); /* Clear the status byte, a valid command was received.*/
191 CMD.command[ 0 ] = temp; /* Put the valid command byte into the first position of the Command[] array.*/
192 COMM.rx_count = 1U; /* Set the receive byte counter to 1.*/
193 while(COMM.rx_count < CMD.cmd_size) /* While the the receive byte counter is less than the command size,*/
194 {
195 CMD.command[ COMM.rx_count ] = Fifo_Get( &RX_FIFO ); /* transfer a byte from the RX_FIFO to the COMMAND array.*/
196 COMM.rx_count = COMM.rx_count + 1U; /* Increment the counter.*/
197 }
198 COMM.processing = true; /* We have a message that needs processing.*/
199 }
200}
